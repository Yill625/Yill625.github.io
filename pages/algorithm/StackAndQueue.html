  const strObj = {
    '(': ')',
    '{': '}',
    '[': ']'
  }
  const arr = []

    const str = s[i]

      arr.push(strObj[str])

        return false
      }
    }
  }
  return !arr.length
}

var dailyTemperatures = function(temperatures) {
  const stack = []
  const res = new Array(temperatures.length).fill(0)
  for (let i = 0; i < temperatures.length; i++) {
